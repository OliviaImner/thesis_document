#PACKAGES
packages <- c(
  "printr",
  "ggthemes",
  "tidyverse",
  "readxl"
)
purrr::walk(packages, library, character.only = TRUE)
rm(packages)
#read data
data <- read_excel(path = "~/Github/thesis_document/interview/interviewScores.xlsx") %>%
  mutate(Nr = parse_factor(Nr, levels = unique(Nr))) %>%
  mutate(Section = if_else(
      Section == "Anvndarbarhet och feedback p prestation", 
      "Anvndarbarhet och feedback\np prestation", Section
    )) %>%
  mutate(Section = parse_factor(Section, levels = unique(Section))) %>%
  mutate(Modell = case_when(
    Modell == "AT" ~ "ATM",
    Modell == "DI" ~ "DIM", 
    Modell == "KL" ~ "KLM"
  ))
figure1a <- data %>%
  ggplot() +
  geom_bar(aes(Nr, Pong, fill = Modell), stat = "identity") +
  facet_wrap(~Section, scales = "free_x") +
  theme_bw() +
  scale_fill_ptol() +
  labs(x = "Frgans identifieringsnummer") +
  theme(legend.position = "top")

figure1a

ggsave(
  plot = figure1a,
  filename = 'figure1a.pdf',
  device = cairo_pdf,
  height = 60,
  width = 93,
  units = "mm"
)

  summarize(
    `Sum of Points` = sum(Pong),
    `Median Points` = median(Pong),
    `Mean Points` = mean(Pong)
  )

figure1b <- data %>%
  ggplot() +
  geom_dotplot(
    aes(Modell, Pong, fill = Modell), 
    binaxis = "y", stackdir = "center", binwidth = 0.1
  ) +
  geom_point(
    data = . %>% 
    group_by(Modell) %>% 
    summarize(`Mean Points` = mean(Pong)),
    aes(Modell, `Mean Points`), shape = 95, size = 10,
    colour = "grey30"
  ) +
  geom_linerange(
    data = . %>%
      group_by(Modell) %>%
      summarize(
        sd.h = mean(Pong) + sd(Pong), 
        sd.l = mean(Pong) - sd(Pong)
      ),
    aes(Modell, ymin = sd.l, ymax = sd.h),
    colour = "grey30"
  ) +
  theme_bw() +
  theme(legend.position = "top") +
  scale_fill_ptol() +
  scale_colour_ptol() +
  guides(fill = FALSE)

ggsave(
  plot = figure1b,

  device = cairo_pdf,
